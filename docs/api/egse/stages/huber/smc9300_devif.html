"""
HUBER Device Interface to the SMC 9300 motor controller.

Author: Rik Huygen
"""

import logging
import socket
import threading
import time

from egse.bits import beautify_binary
from egse.command import ClientServerCommand
from egse.settings import Settings

logger = logging.getLogger(__name__)

huber_settings = Settings.load("Huber Controller")


class HuberError(Exception):
    pass


class HuberSMC9300Command(ClientServerCommand):
    def get_cmd_string(self, *args, **kwargs) -> str:
        out = super().get_cmd_string(*args, **kwargs)
        return out + "\r\n"


# These definitions should be loaded from the huber.yaml file!

cmd_q_system_info      = HuberSMC9300Command(name='get_system_info',   cmd='?')
cmd_q_version          = HuberSMC9300Command(name='get_version',       cmd='?v')
cmd_q_configuration    = HuberSMC9300Command(name='get_configuration', cmd='?conf{axis}')
cmd_q_state            = HuberSMC9300Command(name='get_state',         cmd='?s{axis}')
cmd_q_status           = HuberSMC9300Command(name='get_status',        cmd='?status{axis}')
cmd_q_error            = HuberSMC9300Command(name='get_error',         cmd='?err{axis}')
cmd_q_pos              = HuberSMC9300Command(name='get_current_position', cmd='?p{axis}')
cmd_q_e                = HuberSMC9300Command(name='get_current_encoder_position', cmd='?e{axis}')
cmd_q_ec               = HuberSMC9300Command(name='get_current_encoder_counter_value', cmd='?ec{axis}')


cmd_ecl                = HuberSMC9300Command(name='ecl',              cmd='ecl{axis}:{mode}')
cmd_goto               = HuberSMC9300Command(name='goto',             cmd='goto{axis}:{position}')

class HuberSMC9300EthernetInterface(object):
    def __init__(self):

        self.modelName = None  # Will be initialized by the getHuberModel() call

        # Basic connection settings, loaded from the configuration YAML file

        self.hostname = huber_settings.HOSTNAME
        self.port = huber_settings.PORT

        # Access-to-the-connection semaphore. Use this to lock/unlock I/O access to the
        # connection (whatever type it is) in child classes.

        self.semaphore = threading.Semaphore()

        self.isConnectionOpen = False

        self._num_axes = huber_settings.NUMBER_OF_AXES

    def setConnectionParameters(self, host="localhost", port=None):
        self.hostname = str(host)
        if port is not None:
            self.port = int(str(port))

    def connect(self):

        # Sanity checks

        if self.isConnectionOpen:
            raise HuberError('Socket is already open')
        if self.hostname in (None, ''):
            raise HuberError('ERROR: hostname not initialized')
        if self.port in (None, 0):
            raise HuberError('ERROR: port number not initialized')

        # Create a new socket instance

        try:
            self.sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            #self.sock.setblocking(1)
            # DON'T set a timeout on the socket, this will not result in the expected behavior.
            #
            # If the timeout is set, this will result in the following problem:
            # DEBUG    234    huber.py  send_direct_command: Sent out to HUBER: 'goto1:0.0\r\n'
            # DEBUG    284    huber.py  wait_until_axis_ready: Wait until axis 1 is ready...
            # DEBUG    191    huber.py  get_response: Sent out to HUBER: '?s1\r\n'
            # DEBUG    198    huber.py  get_response: Receiving data after cmd=''?s1\r\n''...
            # WARNING  263    huber.py  wait_for_response: Socket timeout error from timed out
            # DEBUG    200    huber.py  get_response: Received from HUBER: b'\r\n'
            #
            #self.sock.settimeout(3)
        except socket.error as e_socket:
            raise HuberError("ERROR: Failed to create socket.") from e_socket

        # Attempt to establish a connection to the remote host

        # FIXME: Socket shall be closed on exception?

        # We set a timeout of 3 sec before connecting and reset to None
        # (=blocking) after the connect. The reason for this is because when no
        # HUBER SMC is available, e.g during testing, the timeout will take about
        # two minutes which is way too long. It needs to be evaluated if this
        # approach is acceptable and not causing problems during production.

        try:
            logger.debug(f'Connecting a socket to host "{self.hostname}" using port {self.port}')
            self.sock.settimeout(3)
            self.sock.connect((self.hostname, self.port))
            self.sock.settimeout(None)
        except ConnectionRefusedError as e_cr:
            raise HuberError(f'ERROR: Connection refused to {self.hostname}') from e_cr
        except TimeoutError as e_timeout:
            raise HuberError(f'ERROR: Connection to {self.hostname} timed out') from e_timeout
        except socket.gaierror as e_gai:
            raise HuberError(f'ERROR: socket address info error for {self.hostname}') from e_gai
        except socket.herror as e_h:
            raise HuberError(f'ERROR: socket host address error for {self.hostname}') from e_h
        except socket.timeout as e_timeout:
            raise HuberError(f'ERROR: socket timeout error for {self.hostname}') from e_timeout
        except OSError as ose:
            raise HuberError(f"ERROR: OSError caught ({ose}).") from ose

        self.isConnectionOpen = True

        # The first thing to receive should be the 'smc 1.2.1093' string.

        response = self.wait_for_response()
        logger.debug(f"After connection, we got '{response}' as a response.")

        # Check that we are connected to a pmac by issuing the "VERSION" command -- expecting the 1.947 in return.
        # If we don't get the right response, then disconnect automatically

        if not self.is_connected():
            raise HuberError(f"Device is not connected, check logging messages for the cause.")

    def disconnect(self):
        """
        Disconnect from the Ethernet connection.

        Raises a HuberError on failure.
        """
        try:
            if self.isConnectionOpen:
                logger.debug(f'Disconnecting from {self.hostname}')
                self.semaphore.acquire()
                self.sock.close()
                self.semaphore.release()
                self.isConnectionOpen = False
        except Exception as e_exc:
            raise HuberError(f"Could not close socket to {self.hostname}") from e_exc

    def is_connected(self):

        if not self.isConnectionOpen:
            return False

        try:
            response = self.get_response(cmd_q_version.get_cmd_string())
        except HuberError as e_huber:
            if len(e_huber.args) >= 2 and e_huber.args[1] == -1:
                logger.error(f'While trying to talk to the device the following exception occured, exception={e_huber}')
                logger.error(f"Most probably the client connection was closed. Disconnectiing...")
                self.disconnect()
                return False
            else:
                logger.error(f'While trying to talk to the device the following exception occured, exception={e_huber}')
                self.disconnect()
                return False
        finally:
            pass

        version = response

        if not version:
            # if the response was not matching the regex then we're not talking to a PMAC!
            logger.error(f'Device did not respond correctly to a "VERSION" command, response={response}. Disconnecting...')
            self.disconnect()
            return False

        return True

    def get_response(self, command: str, shouldWait=True):
        """
        Send a single command to the controller and block until a response from the controller.

        Arguments:  command (str): is the command to be sent

                    shouldWait (bool, optional, default True): whether to wait on the semaphore.
                        This should normally be left default. If we have acquired the semaphore
                        manually, then specify shouldWait = False (and don't forget to release
                        the semaphore eventually).

        Returns:    response (str): is either a string returned by the PMAC (on success), or
                    an error message (on failure)

        Exception:  HuberError when there was an I/O problem during comm with the PMAC or the
                    response does not have recognised terminators.

                    Note that PMAC may still return an "ERRxxx" code; this function will still
                    consider that a successful transmission.

        """

        try:
            if shouldWait:
                self.semaphore.acquire()

            # Attempt to send the complete command to HUBER

            self.sock.sendall(command.encode())
            # logger.debug(f'Sent out to HUBER: {command!r}')

            # Give the SMC time to fill the buffer
            # time.sleep(0.001)

            # wait for, read and return the response from HUBER (will be at most TBD chars)

            # logger.debug (f"Receiving data after cmd='{command!r}'...")
            response = self.wait_for_response()
            # logger.debug(f'Received from HUBER: {response!r}')

            return response.decode()[:-2]

        except socket.timeout as e_timeout:
            raise HuberError('Socket timeout error') from e_timeout
        except socket.error as e_socket:
            # Interpret any socket-related error as an I/O error
            raise HuberError('Socket communication error.') from e_socket
        finally:
            if shouldWait:
                self.semaphore.release()

    def send_direct_command(self, command: str, shouldWait=False):
        """
        Send a single command to the controller.

        :param str command: is the command string to be sent

        :param bool shouldWait: whether to wait on the semaphore. Default is False
                        This should normally be left default. If we have acquired the semaphore manually,
                        then specify shouldWait = False (and don't forget to release the semaphore eventually).

        :raises: HuberError when there was an I/O problem during communication with the HUBER.

        """

        try:
            if shouldWait:
                self.semaphore.acquire()

            # Attempt to send the complete command to HUBER

            self.sock.sendall(command.encode())
            # logger.debug(f'Sent out to HUBER: {command!r}')

            # Give the SMC time to start a new command: see issue #1209
            time.sleep(0.5)

        except socket.timeout as e_timeout:
            raise HuberError('Socket timeout error') from e_timeout
        except socket.error as e_socket:
            # Interpret any socket-related error as an I/O error
            raise HuberError('Socket communication error.') from e_socket
        finally:
            if shouldWait:
                self.semaphore.release()

    def wait_for_response(self):
        n_total = 0
        buf_size = 2048

        try:
            for idx in range(100):
                # Give the SMC time to fill the buffer
                # time.sleep(0.05)
                data = self.sock.recv(buf_size)
                n = len(data)
                # logger.debug (f"recv: {data.decode()!r}")
                n_total += n
                if n < buf_size:
                    break
        except socket.timeout as e_timeout:
            logger.warning(f"Socket timeout error from {e_timeout}")
            return b'\r\n'

        # logger.debug (f"Total number of bytes received is {n_total}, idx={idx}")

        return data

    def wait_until_axis_ready(self, axis):
        """
        Send the given axis a status query and wait for the response.

        This function will return only when one of the following states has been reached:

        1. The LIMIT- switch was activated
        2. The LIMIT+ switch was activated
        3. The end position was reached, axis ready (bit 0) & controller ready (bit 7)

        :return: 0 on success, i.e. position is reached
        :return: +1 when LIMIT+ switch is reached
        :return: -1 when LIMIT- switch is reached
        """
        logger.debug(f"Wait until axis {axis} is ready...")
        while True:
            time.sleep(0.5)
            response = self.get_response(cmd_q_state.get_cmd_string(axis))
            # response is e.g. '1:<status>;'
            # cut '1:' and ';'
            response = response[2:-1]
            status = int(response)

            logger.debug(f"Status for axis {axis}: {beautify_binary(status)} ({status})")

            if status & 4 == 4:
                logger.debug ("LIMIT- active!")
                rv = -1
                break

            if status & 8 == 8:
                logger.debug ("LIMIT+ active!")
                rv = +1
                break

            if status & 512 == 512:
                logger.error(f"oscillation positioning error (encoder)")

            if status & 129 == 129:
                # exit loop when position is reached
                # status bit 0 and bit 7 indicate 'axis ready' AND 'controller ready' (i.e. all axes stopped)
                rv = 0
                break

            if status & 16384:
                response = self.get_response(cmd_q_error.get_cmd_string(axis))
                logger.error(f"ERROR on axis {axis}: {response}")
                self.send_direct_command(cmd_cerr.get_cmd_string(axis))


        # The sleep of 0.5s here is empirically determined because tests failed randomly because the position was
        # apparently not completely reached. We also noticed an oscilation when arriving at a certain position.
        # A daignostic test revealed that the status sometimes changes bit 0 and bit 7 indicating the axis/controller
        # ready before the mechanism actually reached the setpoont. We need about 0.2s before the actual value is
        # reached, 0.5 seconds is a safe wait time.

        time.sleep(0.5)

        return rv

    # FIXME: Socket shall be closed on exception?

    # We set a timeout of 3 sec before connecting and reset to None
    # (=blocking) after the connect. The reason for this is because when no
    # HUBER SMC is available, e.g during testing, the timeout will take about
    # two minutes which is way too long. It needs to be evaluated if this
    # approach is acceptable and not causing problems during production.

    try:
        logger.debug(f'Connecting a socket to host "{self.hostname}" using port {self.port}')
        self.sock.settimeout(3)
        self.sock.connect((self.hostname, self.port))
        self.sock.settimeout(None)
    except ConnectionRefusedError as e_cr:
        raise HuberError(f'ERROR: Connection refused to {self.hostname}') from e_cr
    except TimeoutError as e_timeout:
        raise HuberError(f'ERROR: Connection to {self.hostname} timed out') from e_timeout
    except socket.gaierror as e_gai:
        raise HuberError(f'ERROR: socket address info error for {self.hostname}') from e_gai
    except socket.herror as e_h:
        raise HuberError(f'ERROR: socket host address error for {self.hostname}') from e_h
    except socket.timeout as e_timeout:
        raise HuberError(f'ERROR: socket timeout error for {self.hostname}') from e_timeout
    except OSError as ose:
        raise HuberError(f"ERROR: OSError caught ({ose}).") from ose

    self.isConnectionOpen = True

    # The first thing to receive should be the 'smc 1.2.1093' string.

    response = self.wait_for_response()
    logger.debug(f"After connection, we got '{response}' as a response.")

    # Check that we are connected to a pmac by issuing the "VERSION" command -- expecting the 1.947 in return.
    # If we don't get the right response, then disconnect automatically

    if not self.is_connected():
        raise HuberError(f"Device is not connected, check logging messages for the cause.")

def get_response(self, command: str, shouldWait=True):
    """
    Send a single command to the controller and block until a response from the controller.

    Arguments:  command (str): is the command to be sent

                shouldWait (bool, optional, default True): whether to wait on the semaphore.
                    This should normally be left default. If we have acquired the semaphore
                    manually, then specify shouldWait = False (and don't forget to release
                    the semaphore eventually).

    Returns:    response (str): is either a string returned by the PMAC (on success), or
                an error message (on failure)

    Exception:  HuberError when there was an I/O problem during comm with the PMAC or the
                response does not have recognised terminators.

                Note that PMAC may still return an "ERRxxx" code; this function will still
                consider that a successful transmission.

    """

    try:
        if shouldWait:
            self.semaphore.acquire()

        # Attempt to send the complete command to HUBER

        self.sock.sendall(command.encode())
        # logger.debug(f'Sent out to HUBER: {command!r}')

        # Give the SMC time to fill the buffer
        # time.sleep(0.001)

        # wait for, read and return the response from HUBER (will be at most TBD chars)

        # logger.debug (f"Receiving data after cmd='{command!r}'...")
        response = self.wait_for_response()
        # logger.debug(f'Received from HUBER: {response!r}')

        return response.decode()[:-2]

    except socket.timeout as e_timeout:
        raise HuberError('Socket timeout error') from e_timeout
    except socket.error as e_socket:
        # Interpret any socket-related error as an I/O error
        raise HuberError('Socket communication error.') from e_socket
    finally:
        if shouldWait:
            self.semaphore.release()

            return False
        else:
            logger.error(f'While trying to talk to the device the following exception occured, exception={e_huber}')
            self.disconnect()
            return False
    finally:
        pass

    version = response

    if not version:

        logger.error(f'Device did not respond correctly to a "VERSION" command, response={response}. Disconnecting...')

def wait_until_axis_ready(self, axis):
    """
    Send the given axis a status query and wait for the response.

    This function will return only when one of the following states has been reached:

    1. The LIMIT- switch was activated
    2. The LIMIT+ switch was activated
    3. The end position was reached, axis ready (bit 0) & controller ready (bit 7)

    :return: 0 on success, i.e. position is reached
    :return: +1 when LIMIT+ switch is reached
    :return: -1 when LIMIT- switch is reached
    """
    logger.debug(f"Wait until axis {axis} is ready...")
    while True:
        time.sleep(0.5)
        response = self.get_response(cmd_q_state.get_cmd_string(axis))
        # response is e.g. '1:<status>;'
        # cut '1:' and ';'
        response = response[2:-1]
        status = int(response)

        logger.debug(f"Status for axis {axis}: {beautify_binary(status)} ({status})")

        if status & 4 == 4:
            logger.debug ("LIMIT- active!")
            rv = -1
            break

        if status & 8 == 8:
            logger.debug ("LIMIT+ active!")
            rv = +1
            break

        if status & 512 == 512:
            logger.error(f"oscillation positioning error (encoder)")

        if status & 129 == 129:
            # exit loop when position is reached
            # status bit 0 and bit 7 indicate 'axis ready' AND 'controller ready' (i.e. all axes stopped)
            rv = 0
            break

        if status & 16384:
            response = self.get_response(cmd_q_error.get_cmd_string(axis))
            logger.error(f"ERROR on axis {axis}: {response}")
            self.send_direct_command(cmd_cerr.get_cmd_string(axis))


    # The sleep of 0.5s here is empirically determined because tests failed randomly because the position was
    # apparently not completely reached. We also noticed an oscilation when arriving at a certain position.
    # A daignostic test revealed that the status sometimes changes bit 0 and bit 7 indicating the axis/controller
    # ready before the mechanism actually reached the setpoont. We need about 0.2s before the actual value is
    # reached, 0.5 seconds is a safe wait time.

    time.sleep(0.5)

    return rv
