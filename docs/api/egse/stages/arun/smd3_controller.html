DEVICE_SETTINGS = Settings.load("Arun SMD3 Controller")

DEVICE_PROTOCOL = Settings.load(filename='smd3.yaml')['Commands']


class Smd3Exception(Exception):

class Smd3Mode:

class Smd3Controller(SerialDevice, Smd3Interface):

    STEP_SIZE_MM        = 5E-3 # step size (5um) but in mm
    MAX_VELOCITY        = 500  # Hz
    POS_DISABLE_STEPS   = -6390
    POS_ENABLE_STEPS    = 32110

    def __init__(self, port='/dev/smd3', baudrate=115200):

        # Load device configuration from the common-egse global config file
        self._port              = DEVICE_SETTINGS.PORT if port is None else port
        self._baudrate          = DEVICE_SETTINGS.BAUDRATE if baudrate is None else baudrate
        self._limit_negative    = None
        self._limit_positive    = None

        # Generate the path to the yaml file
        self.storage_path = os.path.abspath("/data/plato1/egse_plato/smd_position.yaml")
        # module_path = os.path.abspath(os.path.dirname(__file__))
        # self.storage_path = os.path.join(module_path, "position.yaml")

        # Create a dict of Command objects for each function
        self._commands = {}
        for name, items in DEVICE_PROTOCOL.items():
            if 'cmd' in items:
                self._commands[name] = Command(name, items['cmd'])

        # Initialize the parent class with the port and baudrate
        super().__init__(port=self._port, baudrate=self._baudrate, terminator='\r\n')

        self.connect()

        # Stop the device (just in case)
        self.stop()
        if self.is_moving():
            logger.warning('device was moving, waiting for stop...')
            while self.is_moving():
                time.sleep(0.1)
            logger.info('device stopped')

        # Load the last known position from yaml and write to the controller.
        # Because the limit switches are unreliable, we store the position after each move.
        # We load this during initialization and hope it is actually correct.
        position = self._load_position()
        self.set_position_absolute_steps(position)

        # Write configuration
        self.set_mode(Smd3Mode.REMOTE) # this seems not necessary
        self.set_enable_limit_switches_global(False)
        self.set_enable_limit_switches_individual('-', False)
        self.set_enable_limit_switches_individual('+', False)
        self.set_external_enable(False)
        self.set_max_velocity(self.MAX_VELOCITY)
        self.set_freewheel_mode(2) # Short phases to ground when not moving.
        self.set_running_current(0.150)
        self.set_acceleration_current(0.150)
        self.set_limit_stop_mode(1) # Set hard stop mode.

    def get_idn(self):
        return self._query(self._commands['get_idn'].get_cmd_string())[2]

    def get_version(self):
        return self._query(self._commands['get_version'].get_cmd_string())[2]

    def get_status(self):
        return int(self._query(self._commands['get_status'].get_cmd_string())[0], 16)

    def get_errors(self):
        return int(self._query(self._commands['get_errors'].get_cmd_string())[1], 16)

    def clear_errors(self):
        self._query(self._commands['clear_errors'].get_cmd_string())

    def is_moving(self):
        return not bool(self.get_status() & 0x0040)

    def get_mode(self):
        return int(self._query(self._commands['get_mode'].get_cmd_string())[2][0])

    def set_mode(self, mode):
        self._query(self._commands['set_mode'].get_cmd_string(mode=mode))

    def get_max_velocity(self):
        return float(self._query(self._commands['get_max_velocity'].get_cmd_string())[2][1])

    def set_max_velocity(self, velocity):
        self._query(self._commands['set_max_velocity'].get_cmd_string(velocity=velocity))

    def get_external_enable(self):
        return bool(int(self._query(self._commands['get_external_enable'].get_cmd_string())[2]))

    def set_external_enable(self, enable):
        self._query(self._commands['set_external_enable'].get_cmd_string(enable=int(enable)))

    def get_position_absolute_steps(self):
        return int(float(self._query(
            self._commands['get_position_absolute_steps'].get_cmd_string())[2]))

    def get_position_absolute_mm(self):
        return float(self.get_position_absolute_steps() * self.STEP_SIZE_MM)

    def zero_position_absolute(self):
        self._query(self._commands['zero_position_absolute'].get_cmd_string())
        self._store_position(0)

    def set_position_absolute_steps(self, position):
        self._query(self._commands['set_position_absolute_steps'].get_cmd_string(position=position))
        self._store_position(position)

    def _home_software_direction(self, direction):

        assert direction in [-1, 1]

        limit_mask = 0x02 if direction == -1 else 0x04 # Create bit mask for limit input in status word.

        # Move into limit. (can't use move_position_absolute_steps before device is homed)
        self._query(self._commands['move_position_absolute_steps'].get_cmd_string(
            position=(1000000 * direction)))

        while (not self.get_status() & limit_mask): # Wait until limit switch is asserted.
            logger.info(f'homing, status = 0x{self.get_status():X} position = {self.get_position_absolute_steps()}')
            time.sleep(1)


        logger.info('limit asserted')

        self.set_max_velocity(30) # Set speed to 30 Hz.

        # Move out of limit. (can't use move_position_absolute_steps before device is homed)
        self._query(self._commands['move_position_absolute_steps'].get_cmd_string(
            position=(-1000000 * direction)))

        while (self.get_status() & limit_mask): # Wait until limit is de-asserted.
            logger.info(f'reversing, status = 0x{self.get_status():X} position = {self.get_position_absolute_steps()}')
            time.sleep(1)

        self.stop() # Stop moving.
        logger.info('limit cleared')

        self.set_max_velocity(self.MAX_VELOCITY) # Restore nominal speed setting.

    def home_software(self):

        logger.info('start homing in positive direction...')
        self._home_limitless_software_direction(1)
        self.zero_position_absolute() # set absolute 0 to negative limit (also stores position to yaml)
        self.limit_positive = 0
        logger.info('homing in positive direction completed')

        logger.info('start homing in negative direction...')
        self._home_limitless_software_direction(-1)
        self.limit_negative = self.get_position_absolute_steps()
        self._store_position(self.limit_negative)
        logger.info('homing in negative direction completed')

    def move_position_absolute_mm(self, position):

    def move_mask_fov(self, enable):
        position = self.POS_ENABLE_STEPS if enable else self.POS_DISABLE_STEPS

    def mask_in_fov(self):

    def _query(self, command):
        """ Uses the parent query function and adds some sanity checks. """

        # print('send:', repr(command))
        response = super().query(command)
        # print('recv:', repr(response))

        if response[-2:] != '\r\n':
            raise ConnectionError(f"Missing termination characters in response: {response}")

        msg_parts = response.strip().split(',')

        if len(msg_parts) < 2:
            raise ConnectionError(f"Reveived malformed response: {response}")

        status_word = msg_parts[0]
        error_word = msg_parts[1]

        if len(msg_parts) >= 3 and msg_parts[2][:3] == 'ERR':
            raise ConnectionError(f"Reveived error response. Error flags: {msg_parts[1]}, " +
                                  f"Error code: {msg_parts[2]}")

        # return a list if multiple return values are received
        if len(msg_parts) == 2:
            return status_word, error_word, None
        elif len(msg_parts) > 3:
            return status_word, error_word, msg_parts[2:]
        else:
            return status_word, error_word, msg_parts[2]

    def _load_position(self):

        try:
            with open(self.storage_path) as f:
                d = yaml.full_load(f)
        except Exception as e:
            logger.warning('could not load last position from yaml')
            return 0

        return d['position']


    def _store_position(self, position):

        d = {'position': position}

        with open(self.storage_path, 'w') as f:
            yaml.dump(d, f)

            return status_word, error_word, msg_parts[2:]
        else:
            return status_word, error_word, msg_parts[2]

    def _load_position(self):

        try:
            with open(self.storage_path) as f:
                d = yaml.full_load(f)
        except Exception as e:
            logger.warning('could not load last position from yaml')
            return 0

        return d['position']


    def _store_position(self, position):

        d = {'position': position}

        with open(self.storage_path, 'w') as f:
            yaml.dump(d, f)

    logger.info('start homing in positive direction...')
    self._home_limitless_software_direction(1)
    self.zero_position_absolute() # set absolute 0 to negative limit (also stores position to yaml)

    self._home_limitless_software_direction(-1)
    self.limit_negative = self.get_position_absolute_steps()
    self._store_position(self.limit_negative)
    logger.info('homing in negative direction completed')
