Module egse.fee

This module defines all functionality with respect to the FEE. It defines the structure of the FEE internal register map and the FEE housekeeping packet. The module also provides the FEE Simulator.

Currently, only the N-FEE functionality is implemented in this module.

Currently, only the N-FEE functionality is implemented in this module.
"""

# The default order of the CCD readout as defined in the N-FEE register (new FPA numbering).

# DEFAULT_CCD_READOUT_ORDER = 0b11100100  # old convention
DEFAULT_CCD_READOUT_ORDER = 0b01001110  # new convention

# The mapping of the CCD number to the binary representation in the N-FEE (new FPA numbering).
# Note that CCD numbering is from 1 to 4, while the binary representation is 2 bit.
# Note that CCD idx is used for accessing a list or array using the ccd number

CCD_BIN_TO_ID = [3, 4, 1, 2]  # convert N-FEE CCD number to CGSE CCD Number
CCD_BIN_TO_IDX = [2, 3, 0, 1]  # convert N-FEE CCD number to array index
CCD_ID_TO_BIN = [0, 0b10, 0b11, 0b00, 0b01]  # CCD_ID is [1-4]
CCD_IDX_TO_BIN = [0b10, 0b11, 0b00, 0b01]  # CCD index is [0-3]


def convert_ccd_order_list(value: List[int]):
    """
    Convert a list of CCD numbers into their binary representation that is used by the
    register map of the N-FEE.

    Examples:
        >>> bin(convert_ccd_order_list([1,2,3,4]))
        '0b1001110'
        >>> bin(convert_ccd_order_list([2,2,4,4]))
        '0b1011111'

    Args:
        value (List): a list of four CCD numbers in the range [1-4]

    Returns:
        The binary representation of the ccd_readout_order.

    """
    if len(value) != 4:
        raise ValueError(f"Expected a list of four CCD numbers, got {value}")
    if min(value) < 1 or max(value) > 4:
        raise ValueError(f"Expected all CCD numbers in the range [1-4], got {value}")
    return sum(CCD_ID_TO_BIN[ccd] << idx * 2 for idx, ccd in enumerate(value))


def convert_ccd_order_value(value: int):
    """
    Convert a binary representation of the CCD readout order (ccd_readout_order) into a list
    of CCD numbers.

    Examples:
        >>> convert_ccd_order_value(228)
        [3, 4, 1, 2]
        >>> convert_ccd_order_value(0x4E)
        [1, 2, 3, 4]

    Args:
        value (int): the CCD readout order as returned by the N-FEE

    Returns:
        A list of CCD numbers.
    """
    return [CCD_BIN_TO_ID[(value >> x) & 0b11] for x in [0, 2, 4, 6]]


# The name of the of the enum class `fee_mode` is deliberately chosen in snake case,
# although PEP8 prescribes the use of CamelCase for class names. The reason for this choice
# is readability because the members of the enum are in all caps anyway.
#
# fee_mode.ON_MODE is more readable than NFEEMode.ON_MODE (IMHO)

# noinspection PyPep8Naming
class fee_side(IntEnum):
    """
    The CCD side, i.e. E- and F-side. The convention for left and right is taken from the N-FEE IRD.

    This fee_side is what is reported by the N-FEE in the data packet type of the packet header.
    """

    # See issue #1046 [https://github.com/IvS-KULeuven/plato-common-egse/issues/1046]

    # The N-FEE IRD indicates that the E-side is left and F-side is right and that is the convention
    # adopted by the Common-EGSE. The N-FEE itself however adopted the CCD ICD from E2V which states
    # that the E-side is right. To cope with this we swap the definition of the N-FEE in our CGSE
    # definition.

    F_SIDE = F = RIGHT_SIDE = 0
    E_SIDE = E = LEFT_SIDE = 1


n_fee_side = f_fee_side = fee_side


# noinspection PyPep8Naming
class sensor_sel(IntEnum):
    """
    Definition for the selection of one or both of the CCD sides as defined in the N-FEE register
    map. Left and right side are assigned as defined in the N-FEE IRD.
    """
    # Swap the E-side and F-side definition with respect to what is in the N-FEE register.
    # This is needed because the FPGA implementation has another convention as the register.

    RESERVED = 0
    F_SIDE = F = RIGHT_SIDE = 2
    E_SIDE = E = LEFT_SIDE = 1
    BOTH_SIDES = BOTH = 3


# noinspection PyPep8Naming
class n_fee_mode(IntEnum):
    """
    Enumeration of the operating modes for the N-FEE.

    ON_MODE = 0x0
    FULL_IMAGE_PATTERN_MODE = 0x1
    WINDOWING_PATTERN_MODE = 0x2

    FULL_IMAGE_MODE = 0x5
    WINDOWING_MODE = 0x6
    PERFORMANCE_TEST_MODE = 0x7
    IMMEDIATE_ON_MODE = 0x8
    PARALLEL_TRAP_PUMPING_1_MODE = 0x9
    PARALLEL_TRAP_PUMPING_2_MODE = 0xA
    SERIAL_TRAP_PUMPING_1_MODE = 0xB
    SERIAL_TRAP_PUMPING_2_MODE = 0xC
