import logging
import yaml
import threading
import rich
from time import time

from egse.settings import Settings
from egse.system import replace_environment_variable

from egse.fdir.fdir_manager import FdirManagerProxy

from egse.alert import is_control_server_active_in_thread, _get_commanding_addresses, send_email
from egse.alert.alert_manager_interface import AlertManagerInterface
from egse.alert.alert_workers import limitAlertMonitoringWorker, csAlertMonitoringWorker

logger = logging.getLogger(__name__)
CTRL_SETTINGS = Settings.load("Alert Manager Control Server")
    
class AlertManagerController(AlertManagerInterface):
    def __init__(self, phase = None):
        self.phase = phase
        self.running = False
        
        self.alerts = {}
        self.alerts['sensors'] = {}
        self.alerts['cs']      = {}
        
        self.monitoring_workers = {}
        self.monitoring_threads = {}
        
        self._configuration = {}
        
        self.values = {}
        self.cs_addresses = {}
        
        self.cs_addresses = _get_commanding_addresses()
        
        threads = [None] * len(self.cs_addresses)
        results = [None] * len(self.cs_addresses)   
         
        for idx, (cs, addr) in enumerate(self.cs_addresses.items()):
            threads[idx] = threading.Thread(target=is_control_server_active_in_thread, args=(addr, 2, results, idx))
            threads[idx].start()   

        for idx, (cs, addr) in enumerate(self.cs_addresses.items()):
            threads[idx].join()     
            self.alerts['cs'][cs] = results[idx]

        
        # Check if FDIR control server is online
        try:
            with FdirManagerProxy() as fdir:
                if fdir.is_cs_connected():
                    logger.info(f"Connected to FDIR manager!")
        except:
            logger.warning(f"Could not connect to FDIR manager. FDIR actions will not be taken if FDIR manager is not online")
        
        if self.phase != 'none':
            self.load_configuration(self.phase)            
            self.start_alert_monitor()

    def connect(self):
        return True
    
    def disconnect(self):
        return True
    
    def reconnect(self):
        return True
    
    def is_connected(self):
        return True
    
    def is_simulator(self):
        return False
    
    def reset_alert_dict(self):
        self.running = False
        
        self.alerts = {}
        self.alerts['sensors'] = {}
        self.alerts['cs']      = {}

        self._configuration = {}
        
        self.values = {}
        self.cs_addresses = {}
    
    def populate_alert_dict(self):
        self.alerts = {}
        
        # Load sensor alerts, if configured
        if self._configuration['alerts']:
            logger.info("Creating limit monitoring worker")
            self.alerts['sensors'] = {f'{name}' : {
                'operational_limits': [False, False],
                'rate_of_change': False,
                'mask': False,
                'setpoint' : False,
                'timeout': time()
            } for name, conf in self._configuration['alerts'].items()}
            self.monitoring_workers['limit'] = limitAlertMonitoringWorker(self._configuration)
        else:
            self.alerts['sensors'] = {}

        # Load CS alerts, if configured
        self.cs_addresses = _get_commanding_addresses()
        if self.cs_addresses:
            logger.info("Creating control server monitoring worker")
            self.alerts['cs']      = {f"{key}" : False for key, addr in self.cs_addresses.items()}
            self.monitoring_workers['cs']    = csAlertMonitoringWorker(self._configuration)
        else:
            self.alerts['cs'] = {}  

        # Create a current value dict with all the alerts from the configuration
        self.values = {f'{name}' : 0  for name, conf in self._configuration['alerts'].items()}        
    
    def get_alerts(self, sensor=None):
        if sensor == None:
            return self._configuration['alerts']
        else:
            alerts = []
            for key, value in self._configuration['alerts'].items():
                if value['sensor'] == sensor:
                    alerts.append(self._configuration['alerts'][key])
            return alerts

    
    def start_alert_monitor(self):
        if self.phase != 'none':
            logger.info("Starting the alert monitoring threads...")
            # Start all monitoring threads
            if 'limit' in self.monitoring_workers:
                self.monitoring_threads['limit'] = threading.Thread(target=self.monitoring_workers['limit'].start, args=(self.alerts['sensors'], self.values,))
            if 'cs' in self.monitoring_workers:
                self.monitoring_threads['cs']    = threading.Thread(target=self.monitoring_workers['cs'].start, args=(self.alerts['cs'], ))

            try:
                for thread in self.monitoring_threads.values():
                    thread.daemon = True
                    thread.start()
            except Exception as exc:
                logger.info(f"Could not start monitoring threads: {exc}")
        else:
            logger.warning("Can not start the alert monitor without configuring a phase")
    
    def stop_alert_monitor(self):
        logger.info("Stopping the alert monitor")
        try:
            for worker in self.monitoring_workers.values():
                worker.stop()
        except Exception as exc:
            logger.info(f"Alert monitor worker could not be stopped: {exc}")
        self.running = False
    
    def load_configuration(self, phase: str):
        self.phase = phase
        logger.info(f"Reconfiguring the alert manager with the {phase} configuration")
        assert phase in ['none', 'warm', 'transition', 'cold']
        if phase != 'none':
            # Stop alert monitor thread
            if self.running:
                logger.info("Stopping alert monitoring thread")
                self.stop_alert_monitor()
            
            # Load new configuration file
            self._storage_path = replace_environment_variable(CTRL_SETTINGS.CONFIGURATION_LOCATION) + f"{phase}_alerts.yaml"
            logger.info(f"Loading alert configuration from: {self._storage_path}")
            
            # Read configuration file
            try:
                with open(self._storage_path) as f:
                    self._configuration = yaml.load(f, Loader=yaml.Loader)
            except Exception as e:
                logger.error(f"Could not load Alert configuration from {self._storage_path}: {e}")
            
            self.populate_alert_dict()
        else:
            self.stop_alert_monitor()
            self.reset_alert_dict()
            
    def device_connection_lost(self, device_code: str, args=None):
        logger.info(f"Connection has been lost to the device with code: {device_code}")
        device_intro = f"The device with code: {device_code} has lost its connection to its control server!\n\n"
        device_text = ""
        try:
            with FdirManagerProxy() as fdir:
                fdir.signal_fdir(f"FDIR_{device_code}_HW", args)
        except Exception as e:
            logger.critical(f"Could not fire FDIR signal: {e}")
            device_text += f"Could not fire FDIR signal: 'FDIR_{device_code}_HW'\n"
        else:
            device_text += f"FDIR signal: 'FDIR_{device_code}_HW' has been fired\n"
                
        
        device_text  += f"Please reconnect the device with code: {device_code}"
        send_email(subject='PLATO device connection lost', message=device_intro+device_text,  recipients=replace_environment_variable(CTRL_SETTINGS.EMAIL_WARNING_RECIPIENTS))
