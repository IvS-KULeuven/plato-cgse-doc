class AlertMonitoringWorker(QObject):
    
    alert_values_status_signal = pyqtSignal(dict)
    alert_sensor_status_signal = pyqtSignal(dict)
    alert_fdir_status_signal   = pyqtSignal(dict)
    alert_phase_change_signal  = pyqtSignal(str)
    alert_status_change_signal = pyqtSignal(bool)
    
    def __init__(self):
        super(AlertMonitoringWorker, self).__init__()
        self.activate = False
        self.just_reconnected = True
        
        self.monitoring_socket = None
        self.is_socket_connected = True
        self.monitoring_timeout = 0.5
        
        self.connect_socket()
        
        self.previous_alert_update_status_signal = {}
        self.previous_sensors = {}
        self.previous_fdir    = {}
        self.previous_values  = {}
        self.previous_status = False
        self.previous_phase = ""
    
    def reset_monitor(self):
        self.previous_sensors = {}
        self.previous_fdir = {}
        self.previous_values = {}
        
        
    def connect_socket(self):
        try:
            transport           = ALERT_CTRL_SETTINGS.PROTOCOL
            hostname            = ALERT_CTRL_SETTINGS.HOSTNAME
            
            monitoring_port     = ALERT_CTRL_SETTINGS.MONITORING_PORT
            monitoring_address  = connect_address(transport, hostname, monitoring_port)
        
            self.monitoring_socket = zmq.Context().socket(zmq.SUB)
            self.monitoring_socket.connect(monitoring_address)
            self.monitoring_socket.setsockopt_string(zmq.SUBSCRIBE, "")
            
            self.monitoring_timeout = 0.5
            
            self.is_socket_connected = True
        except:
            self.is_socket_connected = False
            
    def stop(self):
        self.monitoring_socket.close()
        self.is_socket_connected = False
        
        self.active = False
        
    def start_process(self):
        self.run()
        
    @pyqtSlot()
    def run(self):
        self.active = True
        timeout = 0
        while self.is_socket_connected and self.active:
            try:
                socket_list, _, exc_list = zmq.select([self.monitoring_socket], [], [], timeout=self.monitoring_timeout)
                if self.monitoring_socket in socket_list:
                    try:
                        pickle_string = self.monitoring_socket.recv()
                    except Exception:
                        raise Exception
                    
                    monitoring_info = pickle.loads(pickle_string)
                    sensors = monitoring_info['sensors']
                    values  = monitoring_info['values']
                    phase   = monitoring_info['phase']
                    running = monitoring_info['running']

                    if phase != self.previous_phase:
                        self.alert_phase_change_signal.emit(phase)
                        
                    self.previous_phase = phase

                    if values != self.previous_values:
                        self.alert_values_status_signal.emit(values)
                    
                    self.previous_values = values
                    
                    if sensors != self.previous_sensors:
                        self.alert_sensor_status_signal.emit(sensors)
                    
                    self.previous_sensors = sensors
                    
                    if running != self.previous_status:
                        self.alert_status_change_signal.emit(running)
                    
                    self.previous_status = running

                        
            except zmq.ZMQError as exc:
                raise exc

class AlertWidget(QGroupBox):
    def __init__(self, alert_name: str, alert_conf: str):
        super().__init__()
        self.name = alert_name
        self.led     = Led(self)
        self.setMaximumHeight(100)
        bold = QFont()
        bold.setBold(True)
        
        label   = QLabel(f"{self.name}")
        label.setFont(bold)
        
        self.value   = QLabel("Value: 0")
        
        self.limit = QLabel("Limit: 0")
        if 'unit' in alert_conf:
            self.unit = QLabel(f"{alert_conf['unit']}")
        
        self.more = ToggleButton(
            width=10,
            height=10,
            name="More alert information",
            status_tip=f"Show more info about {self.name} alert",
            selected=get_resource(":/icons/more.svg"),
            not_selected=get_resource(":/icons/arrow-up.svg"),
        )

        if 'mask' in alert_conf:
            self.limit.hide()
        
        if 'setpoint' in alert_conf:
            self.limit.hide()
        
        self.layout = QGridLayout()
        
        self.layout.addWidget(self.led, 0, 2, 4, 2)
        self.layout.addWidget(label, 0, 3, 1, 6)
        self.layout.addWidget(self.value, 1, 3, 1, 6)
        if 'unit' in alert_conf:
            self.layout.addWidget(self.unit, 1, 5)
        self.layout.addWidget(self.limit, 3, 3, 1, 6)
        
        self.setLayout(self.layout)

    
class AlertWindow(QGroupBox):
    def __init__(self, configuration = None):
        super().__init__()

        self.configuration = configuration
        if configuration:
            self.layout = QVBoxLayout()
            self.layout.setAlignment(Qt.AlignTop)
            self.sensor_layout = QVBoxLayout()
            self.fdir_layout   = QVBoxLayout()
            self.fdir_widgets = []
            self.alert_widgets = []
            if configuration['alerts']:
                self.alert_widgets = [[None, False] for _ in range(len(configuration['alerts']))]
                for idx, (name, conf) in enumerate(configuration['alerts'].items()):
                    self.alert_widgets[idx][0] = AlertWidget(alert_name=f"{name}", alert_conf=conf)
                    
                    if 'operational_limit' in conf:
                        if 'min' in conf['operational_limit'] and 'max' in conf['operational_limit']:
                            self.alert_widgets[idx][0].limit.setText(f"Min: {conf['operational_limit']['min']}, Max: {conf['operational_limit']['max']}")
                        elif 'min' in conf['operational_limit']:
                            self.alert_widgets[idx][0].limit.setText(f"Min: {conf['operational_limit']['min']}")
                        elif 'max' in conf['operational_limit']:
                            self.alert_widgets[idx][0].limit.setText(f"Max: {conf['operational_limit']['max']}")
                    elif 'rate_of_change' in conf:
                        self.alert_widgets[idx][0].limit.setText(f"Limit: {conf['rate_of_change']['limit']}")
                    
                    self.sensor_layout.addWidget(self.alert_widgets[idx][0])
                self.layout.addLayout(self.sensor_layout)

            if not configuration['alerts'] and not configuration['fdir']:
                logger.error(f"No alerts found")
                sys.exit()
            
            self.setLayout(self.layout)

class AlertUIView(QMainWindow, Observable):
    def __init__(self):
        super().__init__()
        self.setGeometry(300, 300, 480, 900)
        self.setWindowTitle("Alert Manager")
        self.initUI()
        
        self.alert_monitoring_thread = QThread(self)
        self.alert_monitoring_worker = AlertMonitoringWorker()
        self.alert_monitoring_worker.moveToThread(self.alert_monitoring_thread)
        
        self.alert_monitoring_worker.alert_phase_change_signal.connect(self.on_phase_change_signal)
        self.alert_monitoring_worker.alert_values_status_signal.connect(self.on_alert_value_status_signal)
        self.alert_monitoring_worker.alert_sensor_status_signal.connect(self.on_alert_sensor_status_signal)
        self.alert_monitoring_worker.alert_status_change_signal.connect(self.on_alert_status_change_signal)
        self.alert_monitoring_thread.started.connect(self.alert_monitoring_worker.start_process)
        self.alert_monitoring_thread.start()

    def configuration_dialog(self):
        sender = self.sender()
        phase = sender.currentText()
        msg = QMessageBox()
        msg.setIcon(QMessageBox.Question)
        msg.setText(f"You are about to configure the alert manager for the {phase} phase")
        msg.setInformativeText(f"Are you sure that you want to configure the alert manager for the {phase} phase?")
        msg.setWindowTitle(f"Configuration confirmation")
        msg.setStandardButtons(QMessageBox.Ok | QMessageBox.Abort)

        # msg.buttonClicked.connect(lambda: )

        retval = msg.exec_()
        if retval == QMessageBox.Ok:
            self.load_new_phase(phase)
        else:
            self.toolbar.cb.setCurrentText()

    def load_configuration(self, phase):
        self._storage_path = replace_environment_variable(ALERT_CTRL_SETTINGS.CONFIGURATION_LOCATION) + f"{phase}_alerts.yaml"
        logger.info(f"Loading alert configuration from: {self._storage_path}")
        
        try:
            with open(self._storage_path) as f:
                configuration = yaml.load(f, Loader=yaml.Loader)
        except Exception as e:
            logger.error(f"Could not load Alert configuration from {self._storage_path}: {e}")    
            sys.exit()
        self.configuration = configuration
        return configuration
    
    def load_new_phase(self, phase):
        if phase != 'none':
            self.load_configuration(phase)
            self.alert_monitoring_worker.previous_phase = phase
            self.actionObservers({"load_new_configuration" : [phase]})
            self.alert_vbox.removeWidget(self.alert_window)
            self.alert_window = AlertWindow(self.configuration)
            self.alert_vbox.insertWidget(0, self.alert_window)
            self.alert_monitoring_worker.reset_monitor()
        else:
            self.alert_monitoring_worker.previous_phase = phase
            self.actionObservers({"load_new_configuration" : [phase]})
            self.alert_vbox.removeWidget(self.alert_window)
            self.alert_window = AlertWindow()
            self.alert_vbox.insertWidget(0, self.alert_window)
            self.alert_monitoring_worker.reset_monitor()
    
    def on_alert_status_change_signal(self, monitoring_info: bool):
        self.toolbar.switch.set_selected(monitoring_info)
    
    def on_phase_change_signal(self, monitoring_info: str):
        phases = ['none', 'warm', 'transition', 'cold']
        self.load_new_phase(monitoring_info)
        self.toolbar.cb.setCurrentIndex(phases.index(monitoring_info))
        self.toolbar.switch.set_selected(False)
    
    def on_alert_value_status_signal(self, monitoring_info: dict):
        for idx, (name, value) in enumerate(monitoring_info.items()):
            self.alert_window.alert_widgets[idx][0].value.setText(f"Value: {float(value):2f}")
        
    def on_alert_sensor_status_signal(self, monitoring_info: dict):
        for idx, (name, state) in enumerate(monitoring_info.items()):
            if any(state['operational_limits']) or state['rate_of_change'] == True or state['mask'] == True or state['setpoint'] == True:
                self.alert_window.alert_widgets[idx][1] = True
                self.alert_window.alert_widgets[idx][0].led.set_color(3)
                self.alert_window.sensor_layout.removeWidget(self.alert_window.alert_widgets[idx][0])
                self.alert_window.sensor_layout.insertWidget(0, self.alert_window.alert_widgets[idx][0])
                self.alert_window.alert_widgets[idx][0].setStyleSheet("""QGroupBox {border: 2px solid red;}""")
            else:
                self.alert_window.alert_widgets[idx][0].led.set_color(1)
                self.alert_window.alert_widgets[idx][1] = False
                self.alert_window.sensor_layout.removeWidget(self.alert_window.alert_widgets[idx][0])
                self.alert_window.sensor_layout.insertWidget(-1, self.alert_window.alert_widgets[idx][0])
                self.alert_window.alert_widgets[idx][0].setStyleSheet('fusion')
    
    def toggle_alertmanager(self):
        sender = self.sender()
        state = sender.is_selected()
        new_state = state
        sender.set_selected(on=new_state)
        if new_state:
            self.actionObservers({'start_alert_monitor':[]})
        else:
            self.actionObservers({"stop_alert_monitor":[]})   
    
    def initUI(self):
        app_frame = QFrame()
        app_frame.setObjectName("AppFrame")
        
        layout = QVBoxLayout()
        
        self.toolbar = ConfigurationWidget()
        
        self.toolbar.cb.activated[str].connect(self.configuration_dialog)
        self.toolbar.switch.clicked.connect(self.toggle_alertmanager)
        
        
        widget_frame = QFrame()
        
        self.alert_vbox = QVBoxLayout()
        self.alert_window = AlertWindow()
        
        self.alert_vbox.addWidget(self.alert_window)
        
        widget_frame.setLayout(self.alert_vbox)
        
        scroll = QScrollArea()
        scroll.setWidget(widget_frame)
        scroll.setWidgetResizable(True)
        scroll.setVerticalScrollBarPolicy(Qt.ScrollBarAlwaysOn)
        scroll.setHorizontalScrollBarPolicy(Qt.ScrollBarAlwaysOff)
        scroll.setFixedWidth(480)
        
        layout.addWidget(self.toolbar)
        layout.addWidget(scroll)
        
        app_frame.setLayout(layout)
        self.setCentralWidget(app_frame)


    def closeEvent(self, close_event: QCloseEvent) -> None:
        self.alert_monitoring_thread.quit()
        self.alert_monitoring_worker.stop()
        

class AlertUIModel:
    def __init__(self, mode):
        self.mode = mode
        
        if self.mode == 'proxy':
            try:
                self.alert_manager = AlertManagerProxy()
            except Exception as exc:
                raise exc
        elif self.mode == 'simulator':
            self.alert_manager = AlertManagerProxy()
        else:
            raise ValueError(f'Unknown type of alert manager implementation passed into the model')

        if self.alert_manager is not None:
            logger.debug(f"Alert manager controller initialized as {self.mode}")    
    
    def load_new_configuration(self, phase: str):
        with AlertManagerProxy() as am:
            am.load_configuration(phase)
    
    def stop_alert_monitor(self):
        with AlertManagerProxy() as am:
            am.stop_alert_monitor()
            
    def start_alert_monitor(self):
        with AlertManagerProxy() as am:
            am.start_alert_monitor()
    
class AlertUIController(Observer):
    def __init__(self, model: AlertUIModel, view: AlertUIView):
        self.model  = model
        self.view   = view
        self.view.addObserver(self)
        
    def update(self, changed_object):

        text = changed_object.text()

        if text == "Reconnect":

            if changed_object.isChecked():

                logger.debug("Reconnecting the Alert manager model.")

                if self.model.reconnect():

                    self.view.set_connection_state(True)

                    if not self.model.has_commands():

                        self.model.load_commands()

                else:
                    self.view.reconnect_action.setChecked(False)
            else:

                logger.debug("Disconnecting the Alert manager model.")
                self.model.disconnect()
                self.view.set_connection_state(False)

        
    def do(self, actions):
        for action, value in actions.items():
            logger.debug(f'do {action} with {value}')
            if action == 'load_new_configuration':
                with AlertManagerProxy() as am:
                    am.load_configuration(value[0])
            if action == 'stop_alert_monitor':
                with AlertManagerProxy() as am:
                    am.stop_alert_monitor()
            if action == 'start_alert_monitor':
                with AlertManagerProxy() as am:
                    am.start_alert_monitor()
                    # am.start_alert_monitor(value[0])

def parse_arguments():
    """
    Prepare the arguments that are specific for this application.
    """
    parser = argparse.ArgumentParser()
    parser.add_argument(
        "--type",
        dest="type",
        action="store",
        choices={"proxy", "simulator"},
        help="Specify Alert implementation you want to connect to.",
        default="proxy",
    )

    args = parser.parse_args()
    return args


def main():
    lock_file = QLockFile(str(Path("~/alert_ui.app.lock").expanduser()))

    args = list(sys.argv)
    app = QApplication(args)
    
    args = parse_arguments()
    app.setWindowIcon(QIcon(str(get_resource(":/icons/alert.svg"))))

    if lock_file.tryLock(100):
        if args.type == 'proxy':
            try:
                proxy = AlertManagerProxy
            except Exception:
                raise Exception

        view = AlertUIView()
        model = AlertUIModel(args.type)
        controller = AlertUIController(model, view)

        view.show()
        sys.exit(app.exec_())
    else:
        error_message = QMessageBox()
        error_message.setIcon(QMessageBox.Warning)
        error_message.setWindowTitle("Error")
        error_message.setText("The Alert GUI application is already running!")
        error_message.setStandardButtons(QMessageBox.Ok)

        return error_message.exec()

if __name__ == '__main__':
    main()

    def __init__(self, mode):
        self.mode = mode

class AlertUIView(QMainWindow, Observable):
    def __init__(self):
        super().__init__()
        self.setGeometry(300, 300, 480, 900)
        self.setWindowTitle("Alert Manager")
        self.initUI()
        
        self.alert_monitoring_thread = QThread(self)
        self.alert_monitoring_worker = AlertMonitoringWorker()
        self.alert_monitoring_worker.moveToThread(self.alert_monitoring_thread)
        
        self.alert_monitoring_worker.alert_phase_change_signal.connect(self.on_phase_change_signal)
        self.alert_monitoring_worker.alert_values_status_signal.connect(self.on_alert_value_status_signal)
        self.alert_monitoring_worker.alert_sensor_status_signal.connect(self.on_alert_sensor_status_signal)
        self.alert_monitoring_worker.alert_status_change_signal.connect(self.on_alert_status_change_signal)
        self.alert_monitoring_thread.started.connect(self.alert_monitoring_worker.start_process)
        self.alert_monitoring_thread.start()

    def configuration_dialog(self):
        sender = self.sender()
        phase = sender.currentText()
        msg = QMessageBox()
        msg.setIcon(QMessageBox.Question)
        msg.setText(f"You are about to configure the alert manager for the {phase} phase")
        msg.setInformativeText(f"Are you sure that you want to configure the alert manager for the {phase} phase?")
        msg.setWindowTitle(f"Configuration confirmation")
        msg.setStandardButtons(QMessageBox.Ok | QMessageBox.Abort)

        # msg.buttonClicked.connect(lambda: )

        retval = msg.exec_()
        if retval == QMessageBox.Ok:
            self.load_new_phase(phase)
        else:
            self.toolbar.cb.setCurrentText()

    def load_configuration(self, phase):
        self._storage_path = replace_environment_variable(ALERT_CTRL_SETTINGS.CONFIGURATION_LOCATION) + f"{phase}_alerts.yaml"
        logger.info(f"Loading alert configuration from: {self._storage_path}")
        
        try:
            with open(self._storage_path) as f:
                configuration = yaml.load(f, Loader=yaml.Loader)
        except Exception as e:
            logger.error(f"Could not load Alert configuration from {self._storage_path}: {e}")    
            sys.exit()
        self.configuration = configuration
        return configuration
    
    def load_new_phase(self, phase):
        if phase != 'none':
            self.load_configuration(phase)
            self.alert_monitoring_worker.previous_phase = phase
            self.actionObservers({"load_new_configuration" : [phase]})
            self.alert_vbox.removeWidget(self.alert_window)
            self.alert_window = AlertWindow(self.configuration)
            self.alert_vbox.insertWidget(0, self.alert_window)
            self.alert_monitoring_worker.reset_monitor()
        else:
            self.alert_monitoring_worker.previous_phase = phase
            self.actionObservers({"load_new_configuration" : [phase]})
            self.alert_vbox.removeWidget(self.alert_window)
            self.alert_window = AlertWindow()
            self.alert_vbox.insertWidget(0, self.alert_window)
            self.alert_monitoring_worker.reset_monitor()
    
    def on_alert_status_change_signal(self, monitoring_info: bool):
        self.toolbar.switch.set_selected(monitoring_info)
    
    def on_phase_change_signal(self, monitoring_info: str):
        phases = ['none', 'warm', 'transition', 'cold']
        self.load_new_phase(monitoring_info)
        self.toolbar.cb.setCurrentIndex(phases.index(monitoring_info))
        self.toolbar.switch.set_selected(False)
    
    def on_alert_value_status_signal(self, monitoring_info: dict):
        for idx, (name, value) in enumerate(monitoring_info.items()):
            self.alert_window.alert_widgets[idx][0].value.setText(f"Value: {float(value):2f}")
        
    def on_alert_sensor_status_signal(self, monitoring_info: dict):
        for idx, (name, state) in enumerate(monitoring_info.items()):
            if any(state['operational_limits']) or state['rate_of_change'] == True or state['mask'] == True or state['setpoint'] == True:
                self.alert_window.alert_widgets[idx][1] = True
                self.alert_window.alert_widgets[idx][0].led.set_color(3)
                self.alert_window.sensor_layout.removeWidget(self.alert_window.alert_widgets[idx][0])
                self.alert_window.sensor_layout.insertWidget(0, self.alert_window.alert_widgets[idx][0])
                self.alert_window.alert_widgets[idx][0].setStyleSheet("""QGroupBox {border: 2px solid red;}""")
            else:
                self.alert_window.alert_widgets[idx][0].led.set_color(1)
                self.alert_window.alert_widgets[idx][1] = False
                self.alert_window.sensor_layout.removeWidget(self.alert_window.alert_widgets[idx][0])
                self.alert_window.sensor_layout.insertWidget(-1, self.alert_window.alert_widgets[idx][0])
                self.alert_window.alert_widgets[idx][0].setStyleSheet('fusion')
    
    def toggle_alertmanager(self):
        sender = self.sender()
        state = sender.is_selected()
        new_state = state
        sender.set_selected(on=new_state)
        if new_state:
            self.actionObservers({'start_alert_monitor':[]})
        else:
            self.actionObservers({"stop_alert_monitor":[]})   
    
    def initUI(self):
        app_frame = QFrame()
        app_frame.setObjectName("AppFrame")
        
        layout = QVBoxLayout()
        
        self.toolbar = ConfigurationWidget()
        
        self.toolbar.cb.activated[str].connect(self.configuration_dialog)
        self.toolbar.switch.clicked.connect(self.toggle_alertmanager)
        
        
        widget_frame = QFrame()
        
        self.alert_vbox = QVBoxLayout()
        self.alert_window = AlertWindow()
        
        self.alert_vbox.addWidget(self.alert_window)
        
        widget_frame.setLayout(self.alert_vbox)
        
        scroll = QScrollArea()
        scroll.setWidget(widget_frame)
        scroll.setWidgetResizable(True)
        scroll.setVerticalScrollBarPolicy(Qt.ScrollBarAlwaysOn)
        scroll.setHorizontalScrollBarPolicy(Qt.ScrollBarAlwaysOff)
        scroll.setFixedWidth(480)
        
        layout.addWidget(self.toolbar)
        layout.addWidget(scroll)
        
        app_frame.setLayout(layout)
        self.setCentralWidget(app_frame)


    def closeEvent(self, close_event: QCloseEvent) -> None:
        self.alert_monitoring_thread.quit()
        self.alert_monitoring_worker.stop()

class AlertWindow(QGroupBox):
    def __init__(self, configuration = None):
        super().__init__()

        self.configuration = configuration
        if configuration:
            self.layout = QVBoxLayout()
            self.layout.setAlignment(Qt.AlignTop)
            self.sensor_layout = QVBoxLayout()
            self.fdir_layout   = QVBoxLayout()
            self.fdir_widgets = []
            self.alert_widgets = []
            if configuration['alerts']:
                self.alert_widgets = [[None, False] for _ in range(len(configuration['alerts']))]
                for idx, (name, conf) in enumerate(configuration['alerts'].items()):
                    self.alert_widgets[idx][0] = AlertWidget(alert_name=f"{name}", alert_conf=conf)
                    
                    if 'operational_limit' in conf:
                        if 'min' in conf['operational_limit'] and 'max' in conf['operational_limit']:
                            self.alert_widgets[idx][0].limit.setText(f"Min: {conf['operational_limit']['min']}, Max: {conf['operational_limit']['max']}")
                        elif 'min' in conf['operational_limit']:
                            self.alert_widgets[idx][0].limit.setText(f"Min: {conf['operational_limit']['min']}")
                        elif 'max' in conf['operational_limit']:
                            self.alert_widgets[idx][0].limit.setText(f"Max: {conf['operational_limit']['max']}")
                    elif 'rate_of_change' in conf:
                        self.alert_widgets[idx][0].limit.setText(f"Limit: {conf['rate_of_change']['limit']}")
                    
                    self.sensor_layout.addWidget(self.alert_widgets[idx][0])
                self.layout.addLayout(self.sensor_layout)

            if not configuration['alerts'] and not configuration['fdir']:
                logger.error(f"No alerts found")
                sys.exit()
            
            self.setLayout(self.layout)
