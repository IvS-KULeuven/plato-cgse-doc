# The rate of change is calculated according to 
def calculate_roc(metric_name: str, window: int):
    timestamp, values = get_housekeeping(hk_name=metric_name, time_window=window)
    deltaY = float(values[-1]) - float(values[0])
    deltaX = float(timestamp[-1]) - float(timestamp[0])
    roc    = deltaY / deltaX
    if time() - (float(timestamp[-1]) - EPOCH_1958_1970) > 20:
        raise Exception(f"HK value {metric_name} is more than 20 seconds old")
    else:
        return roc

def _get_commanding_addresses():
    processes = {}
    commanding_addresses = {}
    with ProcessManagerProxy() as pm:
        core    = pm.get_core()
        devices = pm.get_devices()

        processes.update(core)
        processes.update(devices)

        for process_name, process_info in processes.items():
            proxy_type  = process_info[0]
            device_args = process_info[1]    

            module_name = proxy_type[7:].rsplit(".", 1)[0]
                
            # PLEASE REMOVE THIS IF YOU SEE THIS!! IT IS ONLY FOR LOCAL DEVELOPMENT!
            if "egse.fdir.gsm" in module_name:
                continue
                
                
            module = importlib.import_module(module_name)
                
            ctrl_settings = module.CTRL_SETTINGS

            transport = ctrl_settings.PROTOCOL
            hostname = ctrl_settings.HOSTNAME

            if module_name == "egse.aeu.aeu" or "egse.tempcontrol.agilent.agilent3497" in module_name:
                name = process_name.split(" ")[1].upper()
                commanding_port = ctrl_settings[name]["COMMANDING_PORT"]
            else:

                commanding_port = ctrl_settings.COMMANDING_PORT
                    
                # This is needed when checking whether or not the CS is active
            commanding_addresses[process_name] = connect_address(transport, hostname, commanding_port)
        return commanding_addresses

def send_email(subject: str, message: str, recipients: str):
    sender      = replace_environment_variable(CTRL_SETTINGS.EMAIL_SENDER)
    server      = replace_environment_variable(CTRL_SETTINGS.EMAIL_SERVER)
    port        = 25

    # Create ssl context
    context = create_default_context()
        
    mail = MIMEMultipart('alternative')
    mail['Subject'] = subject
    mail['From']    = sender
    mail['To']      = recipients
        
    body = message
    body += "\n\nKind regards,\n" \
            "Your friendly PLATO notification bot\n\n" \
            "If I am not working as expected, please contact: s.n.gomashie@sron.nl"
        
    mail.attach(MIMEText(body, 'plain'))
        
    with SMTP(server, port) as smtp:
        smtp.starttls(context=context)
        smtp.sendmail(mail['From'], mail['To'].split(','), mail.as_string())
        smtp.quit()
        # logger.info(f"An alert was triggered and an email was send to {recipients}")
        # print(f"Email was send to {recipients}")
