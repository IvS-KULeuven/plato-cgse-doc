import logging
import rich
import threading

from copy import deepcopy
from time import sleep, time

from egse.settings import Settings
from egse.hk import get_housekeeping
from egse.alert import calculate_roc
from egse.alert.gsm.beaglebone import BeagleboneProxy
from egse.system import EPOCH_1958_1970, replace_environment_variable
from egse.fdir.fdir_manager import FdirManagerProxy

logger = logging.getLogger(__name__)

CTRL_SETTINGS = Settings.load("Alert Manager Control Server")

class limitAlertMonitoringWorker():
    def __init__(self, config):
        logger.info("Limit alert monitoring worker is initializing...")
        
        # seperate sensor alerts and fdir alerts
        self.configuration = config
        # Retrieve the commanding ports of all control servers in the SETUP
        
        # Initialize previous state
        if self.configuration['alerts']:
            
            self._prev_alerts = {f'{name}' : {
                'operational_limits': [False, False],
                'rate_of_change': False,
                'mask': False,
                'setpoint' : False,
                'timeout': None
            } for name, conf in self.configuration['alerts'].items()}
            
        else:
            self._prev_alerts = {}

    def start(self, cur_alerts, cur_value):
        self.cur_alerts = cur_alerts
        self.cur_value = cur_value
        self.run()
    
    def stop(self):
        self.active=False
        pass
            
    def run(self):
        
        self.active = True
        
        while self.active:
            limit_intro = "A limit was exceeded!\nBelow you can find which limit was exceeded and what corrective actions can be taken.\n\nExceeded limit(s):\n"
            limit_text  = ""
            fdir_text   = ""
            actions     = "Corrective actions:\n"
            
            start_time = time()
            # Check the limits for each sensor in the configuration file
            for alert, configuration in self.configuration['alerts'].items():
                try:
                    
                    triggered = None
                    
                    # Retrieve the values within a specific window, if the rate of change needs to be evaluated
                    if 'rate_of_change' in configuration:
                        value = calculate_roc(configuration['sensor'], configuration['rate_of_change']['time_window'])
                        
                        roc_exceeded = True if float(value) > float(configuration['rate_of_change']['limit']) else False
                        self.cur_alerts[alert]['rate_of_change'] = roc_exceeded

                        
                    # Retrieve the last value and compare to the limits, if the operational limit needs to be evaluated
                    elif 'operational_limit' in configuration:
                        timestamp, value = get_housekeeping(hk_name=f'{configuration["sensor"]}')

                        if time() - (timestamp - EPOCH_1958_1970) > 20:
                            raise Exception(f"HK value {configuration['sensor']} is more than 20 seconds old")
                        if 'min' in configuration['operational_limit']:
                            min_limit_exceeded = True if float(configuration['operational_limit']['min'] > float(value)) else False
                            self.cur_alerts[alert]['operational_limits'][0] = min_limit_exceeded
                        if 'max' in configuration['operational_limit']:
                            max_limit_exceeded = True if float(configuration['operational_limit']['max'] < float(value)) else False
                            self.cur_alerts[alert]['operational_limits'][1] = max_limit_exceeded
                            
                    # Retieve the last value and mask it to evaluate a specific flag, if it needs to be evaluated
                    elif 'mask' in configuration:
                        timestamp, value = get_housekeeping(f'{configuration["sensor"]}')
                        
                        if time() - (timestamp - EPOCH_1958_1970) > 20:
                            raise Exception(f"HK value {configuration['sensor']} is more than 20 seconds old")
                        flag = True if (int(value) >> configuration['mask']) & 0b1 else False
                        self.cur_alerts[alert]['mask'] = flag
                        
                    # Retrieve the last value and compare it to the configured setpoint, if it needs to be evaluated
                    elif 'setpoint' in configuration:
                        timestamp, value = get_housekeeping(f'{configuration["sensor"]}')
                        
                        if time() - (timestamp - EPOCH_1958_1970) > 20 or time() - (int(sp_timestamp) - EPOCH_1958_1970) > 20:
                            raise Exception(f"HK value {configuration['sensor']} is more than 20 seconds old")
                        if configuration['setpoint']['value']:
                            sp_timestamp = time() + EPOCH_1958_1970
                            sp_limit = configuration['setpoint']['value']
                        elif configuration['setpoint']['sensor']:
                            sp_timestamp, sp_limit = get_housekeeping(f'{configuration["setpoint"]["sensor"]}')
                        setpoint_exceeded = False if (float(sp_limit) - configuration['setpoint']['offset']) <= float(value) <= (float(sp_limit) + configuration['setpoint']['offset']) else True
                        self.cur_alerts[alert]['setpoint'] = setpoint_exceeded
                    
                    triggered = roc_exceeded or min_limit_exceeded or max_limit_exceeded or flag or setpoint_exceeded
                    
                    if triggered:
                        if self._prev_alerts[alert]['timeout'] != None:
                            configured_timeout = configuration['timeout'] if configuration['timeout'] else 5
                            if (time() - self._prev_alerts[alert]['timeout'] > configured_timeout):
                                self.cur_value[alert] = value
                                if self.cur_alerts[alert] != self._prev_alerts[alert]:
                                    if any(self.cur_alerts[alert]):
                                        if any(self.cur_alerts[alert]['operational_limits']):
                                            limit_text += f"\t-  {alert} exceeded its operational limits of {configuration['operational_limit']['min']}, {configuration['operational_limit']['max']}: {value}\n"
                                            actions += f"\t-  ({alert}): {configuration['actions']}\n"
                                        if self.cur_alerts[alert]['rate_of_change']:
                                            limit_text += f"\t-  {alert} has exceeded its RoC limit of {configuration['rate_of_change']['limit']}. Current RoC: {value}\n"     
                                            actions += f"\t-  ({alert}): {configuration['actions']}\n"
                                        if self.cur_alerts[alert]['mask']:
                                            limit_text += f"\t-  {alert} is {bool(configuration['mask'])}\n"
                                            actions += f"\t-  ({alert}): {configuration['actions']}\n"
                                        if self.cur_alerts[alert]['setpoint']:
                                            limit_text += f"\t-  {alert} was larger than {configuration['setpoint']['offset']}. Current value: {value}, Setpoint: {sp_limit}\n"
                                            actions += f"\t-  ({alert}): {configuration['actions']}\n"
                                            
                                        # Signal FDIR Manager if there is an associated FDIR code
                                        if 'fdir_code' in configuration:
                                            fdir_text += f"FDIR code '{configuration['fdir_code']}' was fired\n"
                                            with FdirManagerProxy() as fdir:
                                                if fdir.is_cs_connected():
                                                    fdir.signal_fdir(configuration['fdir_code'])
                                                else:
                                                    logger.critical(f"Could not pass FDIR code: {configuration['fdir_code']}. Please check if FDIR manager is online!")
                                    self._prev_alerts[alert] = deepcopy(self.cur_alerts[alert])
                        else:
                            self._prev_alerts[alert]['timeout'] = time()
                    else:
                        self._prev_alerts['timeout'] = None     
                                                       
                except Exception as exc:
                    logger.warning(f"{exc}")

            # Only send a new alert if the state has actually changed            
            if limit_text != "":
                if any(any(alert) for alert in self.cur_alerts.values()):
                    send_email(subject="PLATO Limit notification",message=limit_intro+limit_text+"\n"+fdir_text+"\n"+actions, recipients=replace_environment_variable(CTRL_SETTINGS.EMAIL_WARNING_RECIPIENTS))
                    send_sms(level=0)
                    # rich.print(limit_intro+limit_text+'\n'+actions)


class csAlertMonitoringWorker():
    def __init__(self, config):

        logger.info("Control server alert monitoring worker is initializing...")
        
        # seperate sensor alerts and fdir alerts
        self.configuration = config
        
        # Retrieve the commanding ports of all control servers in the SETUP
        self.cs_addresses = _get_commanding_addresses()
        
        # Initialize previous state
        self._prev_alerts = {f"{key}" : False for key, addr in self.cs_addresses.items()}

    def start(self, cur_alerts):
        self.cur_alerts = cur_alerts
        self.run()
    
    def stop(self):
        self.active=False
        pass
            
    def run(self):
        
        self.active = True
        
        while self.active:
            cs_intro    = "A control server is offline!\nBelow you can find which control server went offline\n\nOffline Control Server(s):\n"
            cs_text     = ""

            try:
                threads = [None] * len(self.cs_addresses)
                results = [None] * len(self.cs_addresses)   
         
                for idx, addr in enumerate(self.cs_addresses.values()):
                    threads[idx] = threading.Thread(target=is_control_server_active_in_thread, args=(addr, 5, results, idx))
                    threads[idx].start()   

                for idx, (cs, addr) in enumerate(self.cs_addresses.items()):
                    threads[idx].join()     
                    self.cur_alerts[cs] = results[idx]
                    
                    if self.cur_alerts[cs] != self._prev_alerts[cs]:
                        if not self.cur_alerts[cs]:
                            cs_text += f"\t- {cs} is offline!"
                
                if cs_text != "":
                    if any([not alert for alert in self.cur_alerts.values()]):
                        # logger.info(f"{cs_text}")
                        send_email(subject="PLATO Control Server notification", message=cs_intro+cs_text, recipients=replace_environment_variable(CTRL_SETTINGS.EMAIL_CS_RECIPIENTS))

                 # Copy latest measurements for comparison in the nexy cycle                           
                self._prev_alerts = deepcopy(self.cur_alerts)
    
            except Exception as exc:
                    logger.error(f"An error has occured: {exc}")
                    quit()
